class Main {
  public ststic void main(String[] args) {
    int[] arr = {5, 4, 1, 3, 2};

    bubbleSort(arr);

    for (int i = 0; i < arr.length; i++) {
      System.out.println("arr[ + i + "]: " + arr[i]);
    }
  }

  static void bubbleSort(int[] arr) {
    int temp = 0;
    for (int i = 0; i < arr.length; i++) {
      for (int j = 1; j < arr.length - i; j++) {
        if (arr[j] > arr[j - 1]) {
          temp = arr[j - 1];
          arr[j - 1] = arr[j];
          arr[j] = temp;
        }
      }
    }
  }
}
