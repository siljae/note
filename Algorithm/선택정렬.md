```java
class Main {
  public static void main(String[] args) {
    int[] arr = {5, 4, 1, 3, 2};
    int len = arr.length;

    selectionSort(arr, len);

    for (int i = 0; i < len; i++) {
      System.out.println("arr[" + i + "]: " + arr[i]);
    }
  }

  static void selectionSort(int[] arr, int len) {
    for (int i = 0; i < len - 1; i ++) {
      int minIndex = i;
      for (int j = i + 1; j < len; j++) {
        if (arr[minIndex] > arr[j]) {
          minIndex = j;
        }
      }
      int temp = arr[i];
      arr[i] = arr[minIndex];
      arr[minIndex] = temp;
    }
  }
}
```
